model Competition  { 
  id    Int     @id @default(autoincrement()) 
  name  String
  location String
  date DateTime
  dateCreated DateTime @default(now())
  dateUpdated DateTime @updatedAt
} 

model User  { 
  id    Int     @id @default(autoincrement()) 
  email String  @unique
  nameFirst  String
  nameSecond  String
  dateOfBirth DateTime
  Team Team?
  Role Role

  dateCreated DateTime @default(now())
  dateUpdated DateTime @updatedAt
} 

model Result{
  id Int @id @default(autoincrement())
  athleteId Int
  competitionId Int
  score Json


  dateCreated DateTime @default(now())
  dateUpdated DateTime @updatedAt
}

enum Role { 
  ATHLETE
  CENTER 
  ADMIN 
} 

enum Team { 
  None
  National
  Club 
}
